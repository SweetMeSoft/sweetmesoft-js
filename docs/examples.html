Authentication with JWT

Using JWT tokens for authenticated requests

Login and Store Token
// Login function
function login(username, password) {
    SweetMeSoft.post({
        url: '/api/auth/login',
        data: { username, password },
        successCallback: (response) => {
            // Store JWT token
            localStorage.setItem('jwt_token', response.token);
            console.log('Login successful');
            
            // Redirect to dashboard
            window.location.href = '/dashboard';
        },
        errorCallback: (error) => {
            alert('Login failed: ' + error.message);
        }
    });
}
Authenticated Requests
// Get JWT token from storage
const token = localStorage.getItem('jwt_token');

// Make authenticated request
SweetMeSoft.get({
    url: '/api/protected/profile',
    jwt: token,
    lang: 'en-US',
    successCallback: (profile) => {
        console.log('Profile loaded:', profile);
        displayProfile(profile);
    },
    errorCallback: (error) => {
        if (error.status === 401) {
            // Token expired, redirect to login
            localStorage.removeItem('jwt_token');
            window.location.href = '/login';
        }
    }
});
Utility Function for Authenticated Requests
// Helper function for authenticated requests
function authenticatedRequest(method, url, data = null) {
    const token = localStorage.getItem('jwt_token');
    
    if (!token) {
        window.location.href = '/login';
        return;
    }
    
    const options = {
        url: url,
        jwt: token,
        lang: localStorage.getItem('user_language') || 'en-US',
        errorCallback: (error) => {
            if (error.status === 401) {
                localStorage.removeItem('jwt_token');
                window.location.href = '/login';
            }
        }
    };
    
    if (data) {
        options.data = data;
    }
    
    return SweetMeSoft[method.toLowerCase()](options);
}

// Usage
authenticatedRequest('GET', '/api/dashboard-data')
    .then(() => console.log('Dashboard loaded'));

jQuery Extensions

Using the extended jQuery methods

Form Validation and Control
// Form validation example
function validateForm() {
    let isValid = true;
    
    // Check required fields
    $('.required').each(function() {
        const value = $(this).val();
        
        if (value.isNullOrEmpty()) {
            $(this).addClass('is-invalid');
            isValid = false;
        } else {
            $(this).removeClass('is-invalid');
        }
    });
    
    // Validate email format
    if (!$('#email').checkValidity()) {
        $('#email').addClass('is-invalid');
        isValid = false;
    }
    
    // Enable/disable submit button
    if (isValid) {
        $('#submitBtn').enable();
    } else {
        $('#submitBtn').disable();
    }
    
    return isValid;
}

// Real-time validation
$('.form-control').on('input', validateForm);

// Format dates for display
function formatDates() {
    const now = new Date();
    
    // Different date formats
    const shortDate = SweetMeSoft.getFormatedDate(now, 'MM/dd/yyyy');
    const longDate = SweetMeSoft.getFormatedDate(now, 'dd/MM/yyyy');
    const withTime = SweetMeSoft.getFormatedDate(now, 'yyyy-MM-dd HH:mm:ss', true);
    
    console.log('Short date:', shortDate);
    console.log('Long date:', longDate);
    console.log('With time:', withTime);
    
    // Update UI elements
    $('#currentDate').text(shortDate);
    $('#lastUpdated').text(withTime);
}

// Validate date inputs
function validateDateInput(dateString) {
    const date = new Date(dateString);
    
    if (SweetMeSoft.isValidDate(date)) {
        return SweetMeSoft.getFormatedDate(date, 'yyyy-MM-dd');
    } else {
        throw new Error('Invalid date format');
    }
}

// String manipulation utilities
function processUserInput() {
    const firstName = $('#firstName').val();
    const lastName = $('#lastName').val();
    
    // Validate inputs
    if (firstName.isNullOrEmpty()) {
        alert('First name is required');
        return false;
    }
    
    if (lastName.isNullOrEmpty()) {
        alert('Last name is required');
        return false;
    }
    
    // Format names
    const formattedFirstName = SweetMeSoft.capitalizeFirstLetter(firstName.toLowerCase());
    const formattedLastName = SweetMeSoft.capitalizeFirstLetter(lastName.toLowerCase());
    
    // Update display
    $('#displayName').text(formattedFirstName + ' ' + formattedLastName);
    
    return true;
}
URL Parameters
// Handle URL parameters
function handleUrlParameters() {
    // Get specific parameters
    const userId = SweetMeSoft.getUrlParameter('user_id');
    const tab = SweetMeSoft.getUrlParameter('tab');
    const filter = SweetMeSoft.getUrlParameter('filter');
    
    // Apply parameters to UI
    if (userId) {
        loadUserData(userId);
    }
    
    if (tab) {
        $(`#${tab}Tab`).tab('show');
    }
    
    if (filter) {
        $('#filterSelect').val(filter).trigger('change');
    }
}

// Call on page load
$(document).ready(handleUrlParameters);

function performLongOperation() {
    // Show loading overlay
    SweetMeSoft.on();
    
    // Simulate long operation
    setTimeout(() => {
        // Perform actual work here
        console.log('Operation completed');
        
        // Hide loading overlay
        SweetMeSoft.off();
    }, 3000);
}

// Multiple operations with proper overlay management
function performMultipleOperations() {
    SweetMeSoft.on(); // First operation
    
    operation1()
        .then(() => {
            SweetMeSoft.on(); // Second operation (overlay stays visible)
            return operation2();
        })
        .then(() => {
            SweetMeSoft.off(); // First off
            SweetMeSoft.off(); // Second off (overlay hides)
        })
        .catch(() => {
            SweetMeSoft.off();
            SweetMeSoft.off();
        });
}
